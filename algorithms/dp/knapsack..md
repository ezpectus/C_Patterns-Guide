## üíª –ö–æ–¥ –Ω–∞ C

```c
#include <stdio.h>
#include <string.h>

#define MAX_N 100
#define MAX_W 1000

int knapsack(int n, int W, int w[], int v[]) {
    int dp[MAX_N + 1][MAX_W + 1];
    memset(dp, 0, sizeof(dp));

    for (int i = 1; i <= n; i++) {
        for (int j = 0; j <= W; j++) {
            if (w[i - 1] > j)
                dp[i][j] = dp[i - 1][j];
            else
                dp[i][j] = (dp[i - 1][j] > dp[i - 1][j - w[i - 1]] + v[i - 1])
                           ? dp[i - 1][j]
                           : dp[i - 1][j - w[i - 1]] + v[i - 1];
        }
    }
    return dp[n][W];
}

// üîß –¢–µ—Å—Ç
void run_test() {
    int w[] = {2, 3, 4, 5};
    int v[] = {3, 4, 5, 6};
    int n = 4;
    int W = 5;

    int result = knapsack(n, W, w, v);
    printf("Max value: %d\n", result); // –û—á—ñ–∫—É–≤–∞–Ω–æ: 7
}

int main() {
    run_test();
    return 0;
}

```
